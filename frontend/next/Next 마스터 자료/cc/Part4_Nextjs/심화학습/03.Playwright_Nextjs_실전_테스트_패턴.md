## 1. 네비게이션 & 라우팅 테스트 🔴

> Next.js의 핵심인 파일 시스템 라우팅이 제대로 작동하는지!

```ts
// tests/navigation.spec.ts
import { test, expect } from "@playwright/test";

test.describe("네비게이션", () => {
  test("메인 → 상품 목록 → 상품 상세 흐름", async ({ page }) => {
    // 1. 홈에서 시작
    await page.goto("/");

    // 2. 네비게이션 바의 "상품" 링크 클릭
    await page.getByRole("link", { name: "상품" }).click();

    // 3. Soft Navigation 확인 (새로고침 없이 URL만 변경!)
    await expect(page).toHaveURL("/products");
    await expect(page.getByRole("heading", { name: "상품 목록" })).toBeVisible();

    // 4. 첫 번째 상품 클릭 → 상세 페이지 이동
    await page.getByText("상세보기").first().click();
    await expect(page).toHaveURL(/\/products\/\d+/);

    // 5. 뒤로 가기 → 목록으로 복귀
    await page.goBack();
    await expect(page).toHaveURL("/products");
  });

  test("존재하지 않는 페이지 → 404 표시", async ({ page }) => {
    await page.goto("/this-does-not-exist");
    await expect(page.getByText(/찾을 수 없|not found/i)).toBeVisible();
  });

  test("동적 라우팅 파라미터가 제대로 전달된다", async ({ page }) => {
    await page.goto("/products/42");
    // 페이지에 ID가 표시되는지 확인
    await expect(page.getByText("42")).toBeVisible();
  });
});
```

## 2. 폼 전송 테스트 (Server Actions) 🔴

> Server Actions로 동작하는 폼을 사용자 관점에서 테스트!

### 기본 폼 테스트

```ts
// tests/product-form.spec.ts
import { test, expect } from "@playwright/test";

test.describe("상품 등록 폼", () => {
  // 매 테스트 전에 등록 페이지로 이동
  test.beforeEach(async ({ page }) => {
    await page.goto("/products/new");
  });

  test("정상적인 상품 등록 → 성공 메시지 + 리다이렉트", async ({ page }) => {
    // 1. 폼 입력
    await page.getByLabel("상품명").fill("아키텍트 키보드");
    await page.getByLabel("가격").fill("150000");
    await page.getByLabel("설명").fill("최고급 기계식 키보드");

    // 2. 제출 버튼 클릭
    await page.getByRole("button", { name: "등록" }).click();

    // 3. 결과 확인: 목록 페이지로 리다이렉트!
    await expect(page).toHaveURL("/products");

    // 4. 새로 등록한 상품이 목록에 보이는지!
    await expect(page.getByText("아키텍트 키보드")).toBeVisible();
  });

  test("빈 폼 제출 → 에러 메시지 표시", async ({ page }) => {
    // 아무것도 입력 안 하고 제출!
    await page.getByRole("button", { name: "등록" }).click();

    // 에러 메시지 확인 (Server Action 검증 결과!)
    await expect(page.getByText(/상품명.*필수|2자 이상/)).toBeVisible();
  });

  test("제출 중 버튼 비활성화 (다중 클릭 방지)", async ({ page }) => {
    await page.getByLabel("상품명").fill("테스트 상품");
    await page.getByLabel("가격").fill("50000");

    const submitButton = page.getByRole("button", { name: "등록" });
    await submitButton.click();

    // 클릭 직후 버튼이 비활성화되는지! (useFormStatus의 pending!)
    await expect(submitButton).toBeDisabled();
  });
});
```

### 검색 폼 테스트

```ts
test("검색어 입력 → URL에 쿼리 스트링 반영", async ({ page }) => {
  await page.goto("/products");

  // 검색창에 입력 후 Enter
  await page.getByPlaceholder("검색어를 입력하세요").fill("노트북");
  await page.keyboard.press("Enter");

  // URL에 ?q=노트북 반영 확인
  await expect(page).toHaveURL(/\?q=%EB%85%B8%ED%8A%B8%EB%B6%81|q=노트북/);

  // 검색 결과에 "노트북"이 포함된 상품만 표시
  const products = page.getByTestId("product-card");
  for (const product of await products.all()) {
    await expect(product).toContainText(/노트북/);
  }
});
```

## 3. 인증 흐름 테스트 🔴

### 로그인 → 보호 페이지 접근

```ts
// tests/auth.spec.ts
import { test, expect } from "@playwright/test";

test.describe("인증 흐름", () => {
  test("로그인 성공 → 대시보드 접근", async ({ page }) => {
    // 1. 로그인 페이지로 이동
    await page.goto("/login");

    // 2. 자격 증명 입력
    await page.getByLabel("이메일").fill("admin@test.com");
    await page.getByLabel("비밀번호").fill("password123");

    // 3. 로그인 버튼 클릭
    await page.getByRole("button", { name: "로그인" }).click();

    // 4. 대시보드로 리다이렉트 확인
    await expect(page).toHaveURL("/dashboard");
    await expect(page.getByText("환영합니다")).toBeVisible();
  });

  test("잘못된 비밀번호 → 에러 메시지", async ({ page }) => {
    await page.goto("/login");
    await page.getByLabel("이메일").fill("admin@test.com");
    await page.getByLabel("비밀번호").fill("wrong-password");
    await page.getByRole("button", { name: "로그인" }).click();

    await expect(page.getByText(/비밀번호.*틀|인증.*실패/)).toBeVisible();
    // 여전히 로그인 페이지에 있는지
    await expect(page).toHaveURL("/login");
  });

  test("미로그인 → 보호 페이지 접근 → 로그인으로 리다이렉트", async ({ page }) => {
    // 로그인 없이 대시보드 직접 접근 시도
    await page.goto("/dashboard");

    // 로그인 페이지로 강제 이동되는지
    await expect(page).toHaveURL(/\/login/);
  });
});
```

### 인증 상태 재사용 (로그인 반복 방지!) 🟡

```ts
// tests/auth.setup.ts — 한 번만 로그인하고 상태 저장!
import { test as setup, expect } from "@playwright/test";

const authFile = "tests/.auth/user.json";

setup("인증 상태 준비", async ({ page }) => {
  // 1. 로그인 수행
  await page.goto("/login");
  await page.getByLabel("이메일").fill("admin@test.com");
  await page.getByLabel("비밀번호").fill("password123");
  await page.getByRole("button", { name: "로그인" }).click();
  await expect(page).toHaveURL("/dashboard");

  // 2. 쿠키/스토리지 상태를 파일로 저장!
  await page.context().storageState({ path: authFile });
});
```

```ts
// playwright.config.ts — 저장된 인증 상태 사용!
export default defineConfig({
  projects: [
    // 1단계: 인증 준비 (한 번만 실행!)
    { name: "setup", testMatch: /.*\.setup\.ts/ },

    // 2단계: 실제 테스트 (저장된 쿠키로 이미 로그인 상태!)
    {
      name: "chromium",
      use: {
        ...devices["Desktop Chrome"],
        storageState: "tests/.auth/user.json",  // 저장된 인증 상태!
      },
      dependencies: ["setup"],  // setup이 먼저 실행!
    },
  ],
});
```

```
이점:
  ❌ 과거: 10개 테스트 × 매번 로그인 = 10번 로그인 (느림!)
  ✅ 현재: 1번 로그인 → 상태 저장 → 나머지 테스트는 즉시 인증됨! ⚡
```

## 4. API Mocking (route.fulfill) 🟡

> 외부 API에 의존하지 않고 테스트! (네트워크 없이도!)

```ts
test("API 응답을 가짜로 주입하여 테스트", async ({ page }) => {
  // API 요청을 가로채서 가짜 응답 주입!
  await page.route("**/api/products", (route) => {
    route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify([
        { id: 1, name: "가짜 상품 A", price: 10000 },
        { id: 2, name: "가짜 상품 B", price: 20000 },
      ]),
    });
  });

  await page.goto("/products");

  // 가짜 데이터가 화면에 표시되는지!
  await expect(page.getByText("가짜 상품 A")).toBeVisible();
  await expect(page.getByText("가짜 상품 B")).toBeVisible();
});

test("API 에러 시 에러 UI가 표시되는지", async ({ page }) => {
  // 500 에러 주입!
  await page.route("**/api/products", (route) => {
    route.fulfill({ status: 500, body: "서버 에러!" });
  });

  await page.goto("/products");

  // error.tsx가 작동하는지!
  await expect(page.getByText(/문제가 발생|다시 시도/)).toBeVisible();
});

test("느린 네트워크 시뮬레이션", async ({ page }) => {
  // 3초 지연 주입! (loading.tsx 테스트!)
  await page.route("**/api/products", async (route) => {
    await new Promise((res) => setTimeout(res, 3000));
    route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify([]),
    });
  });

  await page.goto("/products");

  // 로딩 UI가 먼저 보이는지!
  await expect(page.getByText(/로딩|loading/i)).toBeVisible();

  // 3초 후 로딩이 사라지는지!
  await expect(page.getByText(/로딩|loading/i)).not.toBeVisible({ timeout: 5000 });
});
```

## 5. Visual Regression (스크린샷 비교) 🟡

```ts
test("상품 목록 페이지 스크린샷 비교", async ({ page }) => {
  await page.goto("/products");

  // 동적 데이터 안정화 대기
  await page.waitForLoadState("networkidle");

  // 전체 페이지 스크린샷 비교
  await expect(page).toHaveScreenshot("products-list.png", {
    maxDiffPixelRatio: 0.01,  // 1% 미만 차이는 통과!
  });
});

test("상품 카드 컴포넌트 스크린샷", async ({ page }) => {
  await page.goto("/products");

  // 특정 요소만 스크린샷!
  const firstCard = page.getByTestId("product-card").first();
  await expect(firstCard).toHaveScreenshot("product-card.png");
});
```

```bash
# 첫 실행: 기준(baseline) 스크린샷 생성
npx playwright test --update-snapshots

# 이후 실행: 기준과 비교하여 차이 감지!
npx playwright test

# 스크린샷 저장 위치: tests/__screenshots__/
```

## 6. beforeEach / afterEach — 테스트 셋업 🔴

```ts
test.describe("상품 CRUD", () => {
  // 매 테스트 전에 실행!
  test.beforeEach(async ({ page }) => {
    await page.goto("/products");
  });

  // 매 테스트 후에 실행! (정리 작업)
  test.afterEach(async ({ page }) => {
    // 예: 테스트 데이터 정리
  });

  test("목록 표시", async ({ page }) => {
    // 이미 /products에 있음!
    await expect(page.getByRole("heading")).toBeVisible();
  });

  test("검색 작동", async ({ page }) => {
    // 이미 /products에 있음!
    await page.getByPlaceholder("검색").fill("키보드");
  });
});
```
