## 1. Infinite Query — 무한 스크롤 🔴

```tsx
import { useInfiniteQuery } from "@tanstack/react-query";
import { useInView } from "react-intersection-observer";

// API: 커서 기반 페이지네이션
async function fetchProducts(cursor: number = 0) {
  const res = await fetch(`/api/products?cursor=${cursor}&limit=20`);
  return res.json() as Promise<{
    items: Product[];
    nextCursor: number | null;  // null이면 마지막 페이지!
  }>;
}

function ProductList() {
  const { ref, inView } = useInView();

  const {
    data,
    fetchNextPage,
    hasNextPage,
    isFetchingNextPage,
    isLoading,
    error,
  } = useInfiniteQuery({
    queryKey: ["products"],
    queryFn: ({ pageParam }) => fetchProducts(pageParam),
    initialPageParam: 0,
    getNextPageParam: (lastPage) => lastPage.nextCursor ?? undefined,
    // lastPage.nextCursor가 null이면 undefined → hasNextPage = false!
  });

  // 화면 바닥에 도달하면 자동 로드!
  useEffect(() => {
    if (inView && hasNextPage && !isFetchingNextPage) {
      fetchNextPage();
    }
  }, [inView, hasNextPage, isFetchingNextPage, fetchNextPage]);

  // 모든 페이지의 아이템을 하나로!
  const allProducts = data?.pages.flatMap(page => page.items) ?? [];

  if (isLoading) return <ProductSkeleton />;
  if (error) return <ErrorMessage error={error} />;

  return (
    <div>
      <div className="grid grid-cols-3 gap-4">
        {allProducts.map(product => (
          <ProductCard key={product.id} product={product} />
        ))}
      </div>

      {/* 무한 스크롤 트리거 */}
      <div ref={ref} className="h-10 flex items-center justify-center">
        {isFetchingNextPage && <Spinner />}
        {!hasNextPage && <p className="text-gray-400">모든 상품을 불러왔습니다</p>}
      </div>
    </div>
  );
}
```

## 2. Optimistic Update — 즉각 반응하는 UI 🔴

> 서버 응답을 기다리지 않고 **UI를 먼저 변경**! 실패하면 롤백.

```tsx
function useToggleLike(productId: number) {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: () => fetch(`/api/products/${productId}/like`, { method: "POST" }),

    // 1. 서버 요청 전에 UI 먼저 변경!
    onMutate: async () => {
      // 진행 중인 쿼리 취소 (충돌 방지)
      await queryClient.cancelQueries({ queryKey: ["product", productId] });

      // 이전 데이터 백업 (롤백용)
      const previous = queryClient.getQueryData<Product>(["product", productId]);

      // 캐시에서 좋아요 상태 즉시 토글!
      queryClient.setQueryData<Product>(["product", productId], old => {
        if (!old) return old;
        return { ...old, isLiked: !old.isLiked, likeCount: old.isLiked ? old.likeCount - 1 : old.likeCount + 1 };
      });

      return { previous };  // 롤백 데이터
    },

    // 2. 실패 시 롤백!
    onError: (err, vars, context) => {
      if (context?.previous) {
        queryClient.setQueryData(["product", productId], context.previous);
      }
    },

    // 3. 성공이든 실패든 서버 데이터로 동기화
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ["product", productId] });
    },
  });
}

// 사용
function LikeButton({ product }: { product: Product }) {
  const toggleLike = useToggleLike(product.id);

  return (
    <button onClick={() => toggleLike.mutate()}>
      {product.isLiked ? "❤️" : "🤍"} {product.likeCount}
    </button>
  );
}
```

## 3. Dependent Query — 순차적 데이터 로드 🟡

> 첫 번째 쿼리의 결과를 두 번째 쿼리에서 사용

```tsx
function UserOrdersPage({ userId }: { userId: number }) {
  // 1단계: 사용자 정보 로드
  const { data: user } = useQuery({
    queryKey: ["user", userId],
    queryFn: () => fetchUser(userId),
  });

  // 2단계: 사용자의 주문 로드 (user가 있을 때만!)
  const { data: orders } = useQuery({
    queryKey: ["orders", user?.id],
    queryFn: () => fetchOrders(user!.id),
    enabled: !!user,  // ⭐ user가 로드될 때까지 이 쿼리는 실행 안 됨!
  });

  // 병렬 로드 vs 순차 로드
  // 병렬: 두 쿼리가 독립적 → Promise.all 효과
  // 순차: enabled: !!previousData → 이전 결과에 의존
}
```

## 4. Prefetch — 미리 데이터 준비 🟡

> 사용자가 클릭하기 전에 데이터를 미리 캐시에 넣어두기!

```tsx
function ProductList({ products }: { products: Product[] }) {
  const queryClient = useQueryClient();

  // 마우스 hover 시 상세 데이터 프리페치!
  const handlePrefetch = (productId: number) => {
    queryClient.prefetchQuery({
      queryKey: ["product", productId],
      queryFn: () => fetchProduct(productId),
      staleTime: 5 * 60 * 1000,  // 5분 캐시
    });
  };

  return (
    <ul>
      {products.map(product => (
        <li
          key={product.id}
          onMouseEnter={() => handlePrefetch(product.id)}
        >
          <Link href={`/products/${product.id}`}>
            {product.name}
          </Link>
        </li>
      ))}
    </ul>
  );
}
// hover → 캐시에 저장 → 클릭 → 즉시 렌더링! (로딩 0초!)
```

## 5. queryKey 설계 전략 🔴

```tsx
// 계층적 키 구조 → 무효화 시 범위 제어!

// 전체 상품
queryKey: ["products"]

// 필터링된 상품 (필터 조건이 다르면 별도 캐시!)
queryKey: ["products", { category: "electronics", sort: "price" }]

// 특정 상품
queryKey: ["products", productId]

// 특정 상품의 리뷰
queryKey: ["products", productId, "reviews"]

// 특정 상품의 리뷰 + 페이지
queryKey: ["products", productId, "reviews", { page: 1 }]

// 무효화 예시:
queryClient.invalidateQueries({ queryKey: ["products"] });
// → 위의 모든 products 관련 캐시가 무효화!

queryClient.invalidateQueries({ queryKey: ["products", productId] });
// → 특정 상품과 그 하위(리뷰 등)만 무효화

// ⭐ 실무 팁: Query Key Factory 패턴
const productKeys = {
  all: ["products"] as const,
  lists: () => [...productKeys.all, "list"] as const,
  list: (filters: Filters) => [...productKeys.lists(), filters] as const,
  details: () => [...productKeys.all, "detail"] as const,
  detail: (id: number) => [...productKeys.details(), id] as const,
};

// 사용
useQuery({ queryKey: productKeys.detail(123), queryFn: ... });
queryClient.invalidateQueries({ queryKey: productKeys.lists() });
```

## 6. Suspense 모드 🟡

```tsx
import { useSuspenseQuery } from "@tanstack/react-query";

// isLoading 체크가 불필요! Suspense가 대신 처리!
function UserProfile({ userId }: { userId: number }) {
  const { data: user } = useSuspenseQuery({
    queryKey: ["user", userId],
    queryFn: () => fetchUser(userId),
  });

  // 여기 도달 = data가 확실히 있음! (undefined 아님!)
  return <h1>{user.name}</h1>;
}

// 부모에서 Suspense + ErrorBoundary로 감싸기
function Page() {
  return (
    <ErrorBoundary FallbackComponent={ErrorFallback}>
      <Suspense fallback={<UserSkeleton />}>
        <UserProfile userId={1} />
      </Suspense>
    </ErrorBoundary>
  );
}
```

## 면접 대비

### 🔴 필수 질문

**Q1. TanStack Query에서 Optimistic Update란?**

> 서버 응답을 기다리지 않고 UI를 먼저 변경하여 즉각적인 반응성을 제공하는 패턴이다. `onMutate`에서 캐시를 직접 수정하고, `onError`에서 실패 시 이전 데이터로 롤백하며, `onSettled`에서 성공/실패와 관계없이 서버 데이터로 최종 동기화한다. 좋아요, 장바구니 담기 등 즉각 피드백이 중요한 인터랙션에 사용한다.

**Q2. queryKey를 어떻게 설계하나요?**

> 계층적 배열 구조로 설계하여 무효화 범위를 제어한다. `["products"]`를 무효화하면 `["products", id]`, `["products", id, "reviews"]` 등 모든 하위 키가 함께 무효화된다. Query Key Factory 패턴으로 키 생성을 함수화하면 타입 안전하게 관리할 수 있다.

┌────────────────────────────────────────────────┐
│        TanStack Query 심화 핵심 요약             │
├────────────────────────────────────────────────┤
│                                                │

│  Prefetch:                                       │
│    hover 시 queryClient.prefetchQuery            │

│                                                │
│  Dependent Query:                                │
│    enabled: !!previousData                      │
│    → 이전 쿼리 결과에 의존하는 순차 로드          │
│                                                │
│  queryKey: 계층적 배열 + Factory 패턴            │
