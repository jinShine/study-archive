## 1. useDebounce ⭐ (검색 최적화)

```tsx
function useDebounce<T>(value: T, delay = 300): T {
  const [debounced, setDebounced] = useState(value);

  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);

  return debounced;
}

// 사용: 타이핑 멈추고 300ms 후에만 API 호출!
const [query, setQuery] = useState("");
const debouncedQuery = useDebounce(query, 500);

useEffect(() => {
  if (debouncedQuery) searchProducts(debouncedQuery);
}, [debouncedQuery]);
```

## 2. useLocalStorage ⭐ (영구 저장)

```tsx
function useLocalStorage<T>(key: string, initialValue: T) {
  const [value, setValue] = useState<T>(() => {
    if (typeof window === "undefined") return initialValue;
    try {
      const stored = localStorage.getItem(key);
      return stored ? JSON.parse(stored) : initialValue;
    } catch { return initialValue; }
  });

  useEffect(() => {
    localStorage.setItem(key, JSON.stringify(value));
  }, [key, value]);

  return [value, setValue] as const;
}

// 사용
const [theme, setTheme] = useLocalStorage("theme", "light");
```

## 3. useMediaQuery ⭐ (반응형)

```tsx
function useMediaQuery(query: string): boolean {
  const [matches, setMatches] = useState(false);

  useEffect(() => {
    const mq = window.matchMedia(query);
    setMatches(mq.matches);
    const handler = (e: MediaQueryListEvent) => setMatches(e.matches);
    mq.addEventListener("change", handler);
    return () => mq.removeEventListener("change", handler);
  }, [query]);

  return matches;
}

// 사용
const isMobile = useMediaQuery("(max-width: 768px)");
const prefersDark = useMediaQuery("(prefers-color-scheme: dark)");
```

## 4. useClickOutside ⭐ (모달/드롭다운 닫기)

```tsx
function useClickOutside<T extends HTMLElement>(callback: () => void) {
  const ref = useRef<T>(null);

  useEffect(() => {
    const handler = (e: MouseEvent) => {
      if (ref.current && !ref.current.contains(e.target as Node)) {
        callback();
      }
    };
    document.addEventListener("mousedown", handler);
    return () => document.removeEventListener("mousedown", handler);
  }, [callback]);

  return ref;
}

// 사용: 모달 바깥 클릭 시 닫기
function Dropdown({ onClose }: { onClose: () => void }) {
  const ref = useClickOutside<HTMLDivElement>(onClose);
  return <div ref={ref}>드롭다운 내용</div>;
}
```

## 5. useCopyToClipboard (클립보드 복사)

```tsx
function useCopyToClipboard() {
  const [copied, setCopied] = useState(false);

  const copy = useCallback(async (text: string) => {
    try {
      await navigator.clipboard.writeText(text);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    } catch { setCopied(false); }
  }, []);

  return { copy, copied };
}

// 사용
const { copy, copied } = useCopyToClipboard();
<button onClick={() => copy("https://example.com")}>
  {copied ? "복사됨! ✅" : "링크 복사"}
</button>
```

## 6. useToggle (불린 토글)

```tsx
function useToggle(initial = false) {
  const [value, setValue] = useState(initial);
  const toggle = useCallback(() => setValue(v => !v), []);
  const setTrue = useCallback(() => setValue(true), []);
  const setFalse = useCallback(() => setValue(false), []);
  return { value, toggle, setTrue, setFalse };
}
```

## 7. useWindowSize (창 크기 감지)

```tsx
function useWindowSize() {
  const [size, setSize] = useState({ width: 0, height: 0 });

  useEffect(() => {
    const handler = () => setSize({ width: window.innerWidth, height: window.innerHeight });
    handler();
    window.addEventListener("resize", handler);
    return () => window.removeEventListener("resize", handler);
  }, []);

  return size;
}
```

## 8. useScrollPosition (스크롤 감지)

```tsx
function useScrollPosition() {
  const [scrollY, setScrollY] = useState(0);
  const [isScrollingUp, setIsScrollingUp] = useState(true);

  useEffect(() => {
    let lastScrollY = window.scrollY;
    const handler = () => {
      const currentY = window.scrollY;
      setIsScrollingUp(currentY < lastScrollY);
      setScrollY(currentY);
      lastScrollY = currentY;
    };
    window.addEventListener("scroll", handler, { passive: true });
    return () => window.removeEventListener("scroll", handler);
  }, []);

  return { scrollY, isScrollingUp };
}

// 사용: 스크롤 시 헤더 숨기기
function Header() {
  const { isScrollingUp } = useScrollPosition();
  return (
    <header style={{ transform: isScrollingUp ? "translateY(0)" : "translateY(-100%)" }}>
      네비게이션
    </header>
  );
}
```

## 9. useInterval (반복 타이머)

```tsx
function useInterval(callback: () => void, delay: number | null) {
  const savedCallback = useRef(callback);

  useEffect(() => { savedCallback.current = callback; }, [callback]);

  useEffect(() => {
    if (delay === null) return;  // null이면 정지!
    const id = setInterval(() => savedCallback.current(), delay);
    return () => clearInterval(id);
  }, [delay]);
}

// 사용: 1초마다 카운트 (null로 멈춤)
const [count, setCount] = useState(0);
const [isRunning, setIsRunning] = useState(true);
useInterval(() => setCount(c => c + 1), isRunning ? 1000 : null);
```

## 10. usePrevious (이전 값 추적)

```tsx
function usePrevious<T>(value: T): T | undefined {
  const ref = useRef<T>();
  useEffect(() => { ref.current = value; });
  return ref.current;
}

// 사용
const [count, setCount] = useState(0);
const prevCount = usePrevious(count);
// "현재: 5, 이전: 4"
```

## 실무 팁: 라이브러리 vs 직접 구현

```
직접 구현:
  ✅ 원리를 이해하고 커스터마이징 가능
  ✅ 번들에 불필요한 코드 없음
  ❌ 엣지 케이스 처리를 직접 해야 함

라이브러리 사용 (usehooks-ts, react-use, ahooks):
  ✅ 엣지 케이스 이미 처리됨
  ✅ SSR 호환, 테스트 완료
  ❌ 불필요한 훅까지 번들에 포함될 수 있음
  → 트리 셰이킹이 되는 라이브러리 선택!

추천: usehooks-ts (TS 기본, 트리 셰이킹 OK)
  npm install usehooks-ts
```
