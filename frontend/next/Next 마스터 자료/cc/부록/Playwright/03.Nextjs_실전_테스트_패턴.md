## 1. 패턴 1: 네비게이션 & 라우팅 테스트 🔴

> 💡 Next.js의 핵심인 "파일 시스템 라우팅"이 제대로 작동하는지 확인!
> 사용자가 링크를 클릭하면 → 올바른 페이지가 뜨는지?
> 뒤로 가기를 누르면 → 이전 페이지로 돌아가는지?

```ts
// tests/navigation.spec.ts
import { test, expect } from "@playwright/test";

test.describe("네비게이션", () => {
  // ─────────────────────────────────
  // 테스트 1: 기본 페이지 이동
  // ─────────────────────────────────
  test("홈 → 상품 목록 → 상품 상세 흐름", async ({ page }) => {
    // 1. 홈에서 시작
    await page.goto("/");

    // 2. "상품" 링크 클릭!
    await page.getByRole("link", { name: "상품" }).click();

    // 3. URL이 /products로 바뀌었는지?
    await expect(page).toHaveURL("/products");

    // 4. 상품 목록 제목이 보이는지?
    await expect(
      page.getByRole("heading", { name: "상품 목록" })
    ).toBeVisible();

    // 5. 첫 번째 상품의 "상세보기" 클릭
    await page.getByText("상세보기").first().click();

    // 6. 상세 페이지 URL 확인 (/products/숫자)
    await expect(page).toHaveURL(/\/products\/\d+/);
    // \d+ = 숫자 1개 이상! /products/1, /products/42 등!

    // 7. 뒤로 가기 → 목록으로 복귀
    await page.goBack();
    await expect(page).toHaveURL("/products");
    // 뒤로 가기해도 목록이 그대로 있는지까지 확인! ✅
  });

  // ─────────────────────────────────
  // 테스트 2: 404 페이지
  // ─────────────────────────────────
  test("존재하지 않는 페이지 → 404 표시", async ({ page }) => {
    // 아무 말이나 적어서 존재하지 않는 주소 접속!
    await page.goto("/asdfghjkl-없는-페이지");

    // not-found.tsx가 작동하는지!
    await expect(
      page.getByText(/찾을 수 없|not found|404/i)
    ).toBeVisible();
    // /i = 대소문자 무시! "Not Found"도 "not found"도 찾음!
  });

  // ─────────────────────────────────
  // 테스트 3: 동적 라우팅 파라미터
  // ─────────────────────────────────
  test("동적 라우팅 [id] 파라미터가 정확히 전달된다", async ({ page }) => {
    await page.goto("/products/42");

    // 페이지에 42라는 ID가 표시되는지!
    await expect(page.getByText("42")).toBeVisible();
  });
});
```

## 2. 패턴 2: 폼 전송 테스트 (Server Actions) 🔴

> 💡 Server Actions로 폼을 제출하면 → 데이터가 저장되고 → 리다이렉트되는지!
> Next.js에서 가장 중요한 테스트 중 하나!

```ts
// tests/product-form.spec.ts
import { test, expect } from "@playwright/test";

test.describe("상품 등록 폼", () => {

  // 매 테스트 전에 등록 페이지로 이동!
  // → 매번 page.goto를 안 써도 돼서 편함!
  test.beforeEach(async ({ page }) => {
    await page.goto("/products/new");
  });

  // ─────────────────────────────────
  // 테스트 1: 정상 등록 → 성공!
  // ─────────────────────────────────
  test("정상 데이터 입력 → 등록 성공 → 목록 페이지 이동", async ({ page }) => {
    // 1. 폼에 데이터 입력
    await page.getByLabel("상품명").fill("아키텍트 키보드");
    await page.getByLabel("가격").fill("150000");

    // 2. 등록 버튼 클릭!
    await page.getByRole("button", { name: /등록|저장/ }).click();
    //  /등록|저장/ = "등록" 또는 "저장"이라는 글자가 있으면 OK!

    // 3. 목록 페이지로 리다이렉트되었는지?
    await expect(page).toHaveURL("/products");

    // 4. 방금 등록한 상품이 목록에 보이는지?
    await expect(page.getByText("아키텍트 키보드")).toBeVisible();
  });

  // ─────────────────────────────────
  // 테스트 2: 빈 폼 제출 → 에러 표시!
  // ─────────────────────────────────
  test("빈 폼 제출 → 에러 메시지 표시", async ({ page }) => {
    // 아무것도 입력 안 하고 바로 제출!
    await page.getByRole("button", { name: /등록|저장/ }).click();

    // Server Action의 유효성 검사 결과로 에러가 뜨는지!
    await expect(
      page.getByText(/필수|2자 이상|입력해/)
    ).toBeVisible();

    // 여전히 등록 페이지에 있어야 함! (리다이렉트 안 됨!)
    await expect(page).toHaveURL("/products/new");
  });

  // ─────────────────────────────────
  // 테스트 3: 다중 클릭 방지 (Pending UI)
  // ─────────────────────────────────
  test("제출 중 버튼 비활성화 (다중 클릭 방지!)", async ({ page }) => {
    // 정상 데이터 입력
    await page.getByLabel("상품명").fill("테스트 상품");
    await page.getByLabel("가격").fill("50000");

    // 제출 버튼 클릭!
    const submitBtn = page.getByRole("button", { name: /등록|저장/ });
    await submitBtn.click();

    // 클릭 직후 버튼이 비활성화(disabled)되는지!
    // useFormStatus의 pending이 작동하는 것을 검증!
    await expect(submitBtn).toBeDisabled();
    // → 비활성화되면 연타해도 한 번만 전송됨! ✅
  });
});
```

### 검색 폼 테스트

```ts
// tests/search.spec.ts
import { test, expect } from "@playwright/test";

test("검색어 입력 → URL 쿼리 스트링 반영", async ({ page }) => {
  // 1. 상품 목록으로 이동
  await page.goto("/products");

  // 2. 검색창 찾기 + 입력 + Enter!
  const searchInput = page.getByPlaceholder("검색어를 입력하세요");
  await searchInput.fill("노트북");
  await searchInput.press("Enter");

  // 3. URL에 ?q=노트북 반영 확인!
  await expect(page).toHaveURL(/[?&]q=/);
  // 한글이 URL 인코딩될 수 있으므로 정규식으로 유연하게!

  // 4. 검색 결과에 "노트북"이 포함된 것만 표시되는지!
  const products = page.getByTestId("product-card");
  const count = await products.count();

  // 모든 상품 카드에 "노트북"이 포함되어 있는지!
  for (let i = 0; i < count; i++) {
    await expect(products.nth(i)).toContainText(/노트북/);
  }
});

test("검색 초기화 → 전체 목록 복원", async ({ page }) => {
  // 검색 상태에서 시작
  await page.goto("/products?q=노트북");

  // 초기화 버튼이나 빈 검색으로 초기화
  const searchInput = page.getByPlaceholder("검색어를 입력하세요");
  await searchInput.clear();
  await searchInput.press("Enter");

  // URL에서 쿼리 스트링이 사라졌는지!
  await expect(page).toHaveURL("/products");
});
```

## 3. 패턴 3: 인증 흐름 테스트 🔴

> 💡 로그인 → 보호 페이지 접근 → 로그아웃 전체 흐름!

### 기본 인증 테스트

```ts
// tests/auth.spec.ts
import { test, expect } from "@playwright/test";

test.describe("인증 흐름", () => {

  // ─────────────────────────────────
  // 테스트 1: 로그인 성공!
  // ─────────────────────────────────
  test("정상 로그인 → 대시보드 접근", async ({ page }) => {
    // 1. 로그인 페이지로 이동
    await page.goto("/login");

    // 2. 이메일 + 비밀번호 입력
    await page.getByLabel("이메일").fill("admin@test.com");
    await page.getByLabel("비밀번호").fill("password123");

    // 3. 로그인 버튼 클릭!
    await page.getByRole("button", { name: "로그인" }).click();

    // 4. 대시보드로 이동되었는지?
    await expect(page).toHaveURL("/dashboard");

    // 5. 환영 메시지가 보이는지?
    await expect(page.getByText(/환영|dashboard/i)).toBeVisible();
  });

  // ─────────────────────────────────
  // 테스트 2: 잘못된 비밀번호!
  // ─────────────────────────────────
  test("잘못된 비밀번호 → 에러 메시지", async ({ page }) => {
    await page.goto("/login");

    await page.getByLabel("이메일").fill("admin@test.com");
    await page.getByLabel("비밀번호").fill("wrong-password");
    await page.getByRole("button", { name: "로그인" }).click();

    // 에러 메시지!
    await expect(
      page.getByText(/비밀번호.*틀|인증.*실패|잘못된/i)
    ).toBeVisible();

    // 여전히 로그인 페이지!
    await expect(page).toHaveURL("/login");
  });

  // ─────────────────────────────────
  // 테스트 3: 미로그인 → 보호 페이지 차단!
  // ─────────────────────────────────
  test("미로그인 시 보호 페이지 접근 → 로그인으로 리다이렉트", async ({ page }) => {
    // 로그인 안 한 채로 대시보드 직접 접근!
    await page.goto("/dashboard");

    // 로그인 페이지로 강제 이동되는지!
    await expect(page).toHaveURL(/\/login/);
  });
});
```

### 인증 상태 재사용 — 매번 로그인 반복 방지! 🟡

> 💡 **비유: 매번 출근할 때마다 면접을 다시 보는 건 미친 짓이다!**
> 한 번 합격(로그인)하면 사원증(쿠키)을 발급받고 다음부턴 사원증만 보여주면 됨!

```ts
// tests/auth.setup.ts — 딱 한 번만 로그인하고 상태 저장!
import { test as setup, expect } from "@playwright/test";

const authFile = "tests/.auth/user.json";  // 사원증(쿠키) 저장 위치!

setup("인증 상태 준비", async ({ page }) => {
  // 1. 로그인 수행
  await page.goto("/login");
  await page.getByLabel("이메일").fill("admin@test.com");
  await page.getByLabel("비밀번호").fill("password123");
  await page.getByRole("button", { name: "로그인" }).click();

  // 2. 로그인 성공 확인
  await expect(page).toHaveURL("/dashboard");

  // 3. 쿠키/스토리지 상태를 파일로 저장!
  await page.context().storageState({ path: authFile });
  // → tests/.auth/user.json 파일에 쿠키가 저장됨!
  // → 다른 테스트에서 이 파일을 읽어서 이미 로그인된 상태로 시작!
});
```

```ts
// playwright.config.ts에 추가!
export default defineConfig({
  projects: [
    // 1단계: 로그인 준비 (딱 한 번만!)
    {
      name: "setup",
      testMatch: /.*\.setup\.ts/,
      // setup이라는 이름의 파일만 실행!
    },

    // 2단계: 실제 테스트 (이미 로그인된 상태!)
    {
      name: "chromium",
      use: {
        ...devices["Desktop Chrome"],
        // ⭐ 저장된 쿠키를 자동으로 불러옴!
        storageState: "tests/.auth/user.json",
      },
      dependencies: ["setup"],
      // setup이 먼저 실행되어야 내가 시작!
    },
  ],
});
```

```
효과:
  ❌ 과거: 10개 테스트 × 매번 로그인 = 로그인 10번! (느림!)
  ✅ 현재: 1번 로그인 → 쿠키 저장 → 나머지 10개는 즉시! ⚡

  시간 절약: 로그인 1번에 3초 × 10개 = 30초 절약!
  안정성: 로그인 과정에서 실패할 확률도 1/10로 줄어듦!
```

## 4. 패턴 4: API Mocking — 가짜 응답 주입! 🟡

> 💡 **비유: 대역 배우를 세우는 것!**
> 실제 API 서버가 없어도, 가짜 응답을 만들어서 테스트!
> "서버가 500 에러를 뱉으면 → error.tsx가 뜨는지?" 같은 것도 테스트 가능!

### 가짜 데이터 주입

```ts
test("API Mock: 가짜 상품 데이터로 테스트", async ({ page }) => {

  // ⭐ page.route(): 특정 URL로 가는 요청을 가로채기!
  await page.route("**/api/products", (route) => {
    // route.fulfill(): 가로챈 요청에 가짜 응답을 돌려줌!
    route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify([
        { id: 1, name: "가짜 상품 A", price: 10000 },
        { id: 2, name: "가짜 상품 B", price: 20000 },
      ]),
    });
  });
  // ** = "아무 도메인이나" (와일드카드!)
  // /api/products로 가는 모든 요청을 가로챔!

  // 페이지에 접속하면 가짜 데이터가 표시됨!
  await page.goto("/products");
  await expect(page.getByText("가짜 상품 A")).toBeVisible();
  await expect(page.getByText("가짜 상품 B")).toBeVisible();
});
```

### 에러 상황 테스트 (error.tsx 작동 확인!)

```ts
test("API 에러 시 에러 UI가 표시되는지", async ({ page }) => {
  // 500 에러를 강제 주입!
  await page.route("**/api/products", (route) => {
    route.fulfill({
      status: 500,
      body: "서버가 터졌다!",
    });
  });

  await page.goto("/products");

  // error.tsx가 작동하는지!
  await expect(
    page.getByText(/문제가 발생|다시 시도|에러/i)
  ).toBeVisible();

  // "다시 시도" 버튼이 있는지!
  await expect(
    page.getByRole("button", { name: /다시 시도/i })
  ).toBeVisible();
});
```

### 느린 네트워크 테스트 (loading.tsx 작동 확인!)

```ts
test("느린 API → 로딩 UI가 먼저 표시되는지", async ({ page }) => {
  // 3초 지연을 강제 주입!
  await page.route("**/api/products", async (route) => {
    // 3초 동안 응답을 안 보냄!
    await new Promise((res) => setTimeout(res, 3000));
    route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify([{ id: 1, name: "상품", price: 1000 }]),
    });
  });

  await page.goto("/products");

  // 로딩 UI가 먼저 보이는지! (loading.tsx!)
  await expect(page.getByText(/로딩|loading/i)).toBeVisible();

  // 3초 후 로딩이 사라지고 실제 콘텐츠가 표시되는지!
  await expect(page.getByText("상품")).toBeVisible({ timeout: 5000 });
  // timeout: 5000 = 최대 5초까지 기다려! (3초 지연 + 여유 2초)
});
```

```
API Mocking이 유용한 상황:

  1. 외부 API가 불안정할 때 → 내 테스트가 남의 서버 탓에 실패하면 안 됨!
  2. 에러 상황을 테스트할 때 → 실제 서버에서 500 에러를 일부러 낼 수 없음!
  3. 느린 네트워크를 테스트할 때 → loading.tsx가 작동하는지!
  4. 데이터를 통제하고 싶을 때 → "정확히 이 데이터가 화면에 뜨는지?"
```

## 5. 패턴 5: Next.js 특화 테스트 총정리 🔴

> Next.js의 특수 파일(loading, error, not-found)을 테스트하는 법!

```ts
// tests/next-features.spec.ts
import { test, expect } from "@playwright/test";

test.describe("Next.js 특수 파일 테스트", () => {

  // loading.tsx 테스트
  test("데이터 로딩 중 스켈레톤 UI가 표시된다", async ({ page }) => {
    // 느린 API 주입
    await page.route("**/api/products", async (route) => {
      await new Promise((res) => setTimeout(res, 2000));
      route.fulfill({ status: 200, body: "[]" });
    });

    await page.goto("/products");

    // 스켈레톤 또는 로딩 UI가 보이는지!
    await expect(
      page.getByText(/로딩|loading/i).or(page.locator("[class*=animate-pulse]"))
    ).toBeVisible();
    // .or() = "둘 중 하나라도 보이면 OK!"
  });

  // error.tsx 테스트
  test("서버 에러 시 에러 바운더리가 작동한다", async ({ page }) => {
    await page.route("**/api/products", (route) => {
      route.fulfill({ status: 500 });
    });

    await page.goto("/products");

    await expect(page.getByText(/문제.*발생|에러/i)).toBeVisible();
    await expect(page.getByRole("button", { name: /다시/i })).toBeVisible();
  });

  // not-found.tsx 테스트
  test("404 페이지가 올바르게 표시된다", async ({ page }) => {
    await page.goto("/존재하지않는주소");

    await expect(page.getByText(/찾을 수 없|404/i)).toBeVisible();
    await expect(page.getByRole("link", { name: /홈|메인/i })).toBeVisible();
  });

  // Soft Navigation 테스트 (깜빡임 없이 이동!)
  test("Link로 이동 시 전체 새로고침이 일어나지 않는다", async ({ page }) => {
    await page.goto("/");

    // 네비게이션 전 스크롤 위치나 상태 기록
    await page.evaluate(() => {
      (window as any).__testMarker = "alive";
    });

    // Link로 이동
    await page.getByRole("link", { name: "상품" }).click();
    await expect(page).toHaveURL("/products");

    // 전체 새로고침이 안 일어났다면 → 마커가 살아있음!
    const marker = await page.evaluate(() => (window as any).__testMarker);
    expect(marker).toBe("alive");
    // 만약 하드 네비게이션(전체 새로고침)이 일어났다면
    // → window가 초기화되어 marker가 undefined!
  });
});
```

## 6. describe와 beforeEach — 테스트 정리하기 🟡

```ts
// test.describe(): 관련 테스트를 그룹으로 묶기!
test.describe("상품 CRUD", () => {

  // test.beforeEach(): 그룹 안의 모든 테스트 전에 실행!
  test.beforeEach(async ({ page }) => {
    await page.goto("/products");
    // → 이제 아래 테스트들에서 page.goto 안 써도 됨!
  });

  test("목록이 표시된다", async ({ page }) => {
    // 이미 /products에 있음!
    await expect(page.getByRole("heading", { name: "상품 목록" })).toBeVisible();
  });

  test("검색이 작동한다", async ({ page }) => {
    // 이미 /products에 있음!
    await page.getByPlaceholder("검색").fill("키보드");
    await page.keyboard.press("Enter");
    await expect(page).toHaveURL(/q=/);
  });

  test("상세 페이지로 이동한다", async ({ page }) => {
    // 이미 /products에 있음!
    await page.getByText("상세보기").first().click();
    await expect(page).toHaveURL(/\/products\/\d+/);
  });
});
```

### 🔴 필수 질문

**Q1. Next.js 프로젝트에서 어떤 것들을 E2E 테스트하나요?**

> 핵심 사용자 흐름을 테스트한다. 첫째, 페이지 네비게이션과 라우팅(Link 클릭, 동적 라우트, 뒤로 가기). 둘째, Server Actions를 사용하는 폼 전송(데이터 입력→제출→리다이렉트→목록 반영). 셋째, 인증 흐름(로그인→보호 페이지 접근→미로그인 시 리다이렉트). Next.js 특수 파일인 loading.tsx, error.tsx, not-found.tsx의 작동도 API Mocking으로 검증한다.

**Q2. API Mocking을 왜 사용하나요?**

> 외부 API에 의존하지 않고 안정적으로 테스트하기 위해서다. `page.route()`로 특정 API 요청을 가로채서 가짜 응답을 주입하면, 서버 없이도 정상/에러/느린 네트워크 상황을 시뮬레이션할 수 있다. 특히 error.tsx(500 에러 주입)나 loading.tsx(응답 지연 주입) 테스트에 필수적이다.
